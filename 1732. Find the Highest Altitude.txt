class Solution {
    public int largestAltitude(int[] gain) {
        int n = gain.length;

        int pre[] = new int[10];
        pre[0] = gain[0];
        int max = 0;

        for(int i=1;i<n;i++){
            pre[i] = pre[i-1] + gain[i];

            max = Math.max(max,pre[i]);
        }
        return max;
    }
}
